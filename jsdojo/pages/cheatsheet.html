JS Cheatsheet

Everything you've learned, one scroll away.

Variable Declaration

var   x = 1;    // function-scoped, hoisted
let   y = 2;    // block-scoped, reassignable
const z = 3;    // block-scoped, no reassign

Primitive Types

typeof "hi"       // "string"
typeof 42        // "number"
typeof true      // "boolean"
typeof undefined // "undefined"
typeof null      // "object" ← bug
typeof Symbol()  // "symbol"

Type Conversion

Number("42")      // 42
Number(true)     // 1
Number(null)     // 0
String(42)       // "42"
Boolean(0)       // false
Boolean("")      // false
Boolean({})      // true

Coercion Traps

"5" + 3          // "53"
"5" - 3          // 2
true && "hello"  // "hello"
false || "hi"   // "hi"
0 ?? "default"  // 0
null ?? "def"   // "def"

Operators

5 === 5        // true (strict)
5 == "5"       // true (loose, avoid)
2 ** 10        // 1024 (exponent)
10 % 3         // 1 (modulo)
x > 10 ? 'big' : 'small'

Functions

function add(a, b) { return a+b; }
const mul = function(a,b) { return a*b; }
const sub = (a, b) => a - b;
const dbl = n => n * 2;
function greet(name = "User") {}

Type Checking

Array.isArray([])        // true
Number.isNaN(NaN)        // true
Number.isNaN("hello")   // false
typeof []                // "object"
[] instanceof Array      // true

Template Literals

const name = "Arjun";
const msg = `Hello ${name}!`;
const calc = `${2 ** 8} ways`;
// Multi-line — no \n needed
const multi = `
  line one
  line two
`;
